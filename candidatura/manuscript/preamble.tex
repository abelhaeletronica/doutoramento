% --- Fonts (ambas de Hermann Zapf — combinam perfeitamente) ---
\usepackage{fontspec}
\setmainfont{Libertinus Serif}[Numbers = OldStyle, Ligatures = TeX]
\setsansfont{Inter}[Scale = MatchLowercase]
\newfontfamily\intersemibold{Inter SemiBold}[Scale = MatchLowercase]
\setmonofont{JetBrains Mono}[Scale = 0.85]

% --- Micro-typography ---
\usepackage{microtype}

% --- Colour ---
\usepackage{xcolor}
\definecolor{accent}{HTML}{1A1A1A}
\definecolor{muted}{HTML}{555555}
\definecolor{linecolor}{HTML}{C8C8C8}

% --- Header block ---
\newcommand{\cvname}[1]{%
  {\centering\fontsize{22}{26}\selectfont\sffamily\textbf{#1}\par}%
}
\newcommand{\cvcontact}[1]{%
  \vspace{0.4em}%
  {\centering\small\sffamily\color{muted}#1\par}%
  \vspace{1.4em}%
  {\color{linecolor}\hrule height 0.5pt}\vspace{1.6em}%
}

% --- Section headings ---
\usepackage{titlesec}
\titleformat{\section}
  {\intersemibold\fontsize{13.5}{16.2}\selectfont\color{accent}}
  {}{0em}{}
\titlespacing{\section}{0pt}{0pt}{0pt}

\titleformat{\subsection}
  {\normalfont\small\sffamily\bfseries}
  {}{0em}{}
\titlespacing{\subsection}{0pt}{1.2em}{0.4em}

% --- List spacing ---
\usepackage{enumitem}
\setlist{nosep, leftmargin=*, topsep=2pt}

% --- Paragraph formatting ---
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.5\baselineskip}

% --- Hyphenation ---
\hyphenpenalty=50
\exhyphenpenalty=50

% --- Widow/orphan control ---
\widowpenalty=10000
\clubpenalty=10000

% --- Proposta header block ---

  \vspace{-0.5\baselineskip}%
}
\newcommand{\docsubtitle}[1]{%
  {\centering\intersemibold\fontsize{14}{16.8}\selectfont#1\par}%
}
\newcommand{\authorname}[1]{%
  {\normalsize\sffamily\color{muted}#1\par}%
}
\newcommand{\authoraffil}[1]{%
  {\parskip=0pt\normalsize\sffamily\color{muted}#1\par}%
  \vspace*{-\baselineskip}%
}
